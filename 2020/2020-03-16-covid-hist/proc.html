Load and wrangle all data

df_all <-
  ## Load all files
  pmap_dfr(
    df_files_all,
    function(file, name) {
      accessed <-
        str_extract(name, "_t[\\d-]*_u") %>%
        str_sub(., 3, -2) %>%
        ymd()
      updated <-
        str_extract(name, "_u[\\d-]*") %>%
        str_sub(., 3) %>%
        ymd

      read_csv(file) %>%
        mutate(
          accessed = accessed,
          updated = updated
        )
    }
  ) %>%
  ## Tidy
  pivot_wider(names_from = var, values_from = val)
